from tkinter import * 
from tkinter.ttk import *

from time import strftime


root = Tk()
root.title("Clock")

def time():
    string = strftime('%H:%M:%S %p')
    label.config(text=string)
    label.after (1000, time)

label=Label(root, font=("ds-digital",100), background="green" , foreground= "white")
label.pack(anchor='center')
time()

mainloop()



from tkinter import *: Imports everything from the basic tkinter module (used to build GUI apps).

from tkinter.ttk import *: Imports themed widgets from the ttk module (used for better styling).

from time import strftime: Imports the strftime function to format the current time into a string.

root = Tk(): Creates the main application window.

root.title("Clock"): Sets the window title to "Clock".

strftime('%H:%M:%S %p'):

%H → Hour (00–23)

%M → Minutes (00–59)

%S → Seconds (00–59)

%p → AM/PM

label.config(text=string): Updates the label's text to show the current time.

label.after(1000, time): Schedules the time() function to run again after 1000 milliseconds (1 second), making the clock tick.

Label(...): Creates a text label in the window.

font=("ds-digital", 100) → Uses "ds-digital" font at size 100.

background="green" → Green background.

foreground="white" → White text.

pack(anchor='center'): Places the label at the center of the window.

Calls the time() function once to start the clock.
